class ResonanceMemory:
    """
    Not a key-value store. Not a vector DB.
    A living web of intention-reaction-resonance triples.
    """
    
    def __init__(self):
        self.intention_nodes = {}      # What the operator meant
        self.reaction_nodes = {}       # What the system did
        self.resonance_edges = {}      # The connection between them
        self.operator_field = {}       # The operator's "cognitive shape"
        
    def record_interaction(self, intention_guess, reaction, operator_response):
        """
        Every interaction creates a triangle:
        - What we thought they meant
        - What we did
        - What they did next (the resonance)
        """
        # The operator's next action reveals more about their
        # true intention than any explicit feedback could.
        
        resonance = self._measure_resonance(
            expected=operator_response if intention_guess.was_correct else None,
            actual=operator_response
        )
        
        # Strengthen or weaken the connection
        if resonance > 0.7:
            self._strengthen(intention_guess, reaction)
        elif resonance < 0.3:
            self._weaken(intention_guess, reaction)
            # AND: create a new hypothesis about what they actually wanted
            self._spawn_alternative_intention(intention_guess, operator_response)
    
    def _measure_resonance(self, expected, actual):
        """
        Resonance is not accuracy. It's the degree to which
        the operator's next move *adopts* or *extends* the
        system's response, versus *rejects* or *reframes* it.
        """
        if expected is None:
            # No prior hypothesis — measure from operator's field
            return self._field_alignment(actual)
        
        # Measure: did they build on it, or start over?
        semantic_overlap = self._semantic_similarity(expected, actual)
        conceptual_extension = self._extension_detection(expected, actual)
        
        return (semantic_overlap * 0.4) + (conceptual_extension * 0.6)

class SoftballEngine:
    """
    When intention is unclear, don't guess.
    Pitch a question that reveals the operator's
    implicit structure.
    """
    
    def __init__(self, memory: ResonanceMemory):
        self.memory = memory
        self.interview_depth = 0  # How many softballs we've pitched
        
    def should_interview(self, prompt: str) -> bool:
        """
        Don't interview if:
        - The prompt is highly specific (clear intention)
        - We've seen this exact pattern before (known resonance)
        - The operator's field is strongly aligned
        
        DO interview if:
        - Ambiguous or abstract language ("better", "fix it")
        - Novel context (new project, new domain)
        - Historical misalignment on similar prompts
        """
        ambiguity = self._measure_ambiguity(prompt)
        novelty = self._measure_novelty(prompt)
        historical_misalignment = self._check_misalignment_history(prompt)
        
        return (ambiguity > 0.5 or novelty > 0.7 or historical_misalignment > 0.6)
    
    def pitch_softball(self, prompt: str) -> str:
        """
        The question is not random. It's shaped by:
        - The operator's historical preference patterns
        - The current project context
        - The operator's known cognitive style
        
        Example: If operator historically prefers physical
        interfaces over visual ones, the softball probes
        that dimension first.
        """
