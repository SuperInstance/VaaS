# tzpro-agent/layer_injector.pyimport osimport time
TZ_IMPORT_DIR = r"C:\ProgramData\TimeZero\Import"
def compile_tzx_layer(layer_name, objects):
    """Compiles custom marks/contours into TimeZero Exchange XML format."""
    xml_header = f"""<?xml version="1.0" encoding="UTF-8"?>
<TZX Exchange="TimeZero" Version="1.0">
  <Layer Name="{layer_name}" Visible="True">"""
    
    body = ""
    for obj in objects:
        if obj['type'] == 'mark':
            body += f"""
    <Position Lat="{obj['lat']}" Lon="{obj['lon']}">
      <Marker Icon="Fish" Color="{obj['color']}" Name="{obj['name']}">
        <Description>{obj['desc']}</Description>
      </Marker>
    </Position>"""
        elif obj['type'] == 'contour':
            # Formats custom contour lines as a multi-vertex boundary track
            body += f"""
    <Boundary Color="{obj['color']}" Type="Line">"""
            for pt in obj['points']:
                body += f'\n      <Vertex Lat="{pt[0]}" Lon="{pt[1]}"/>'
            body += "\n    </Boundary>"
            
    xml_footer = "\n  </Layer>\n</TZX>"
    return xml_header + body + xml_footer
def inject_profile_to_vessel(layer_name, objects):
    """Generates and hot-swaps the layer file into the TZPro import engine."""
    tzx_data = compile_tzx_layer(layer_name, objects)
    temp_filename = f"inject_{int(time.time())}.tzx"
    target_path = os.path.join(TZ_IMPORT_DIR, temp_filename)
    
    # Write to local directory; TZPro parses and sweeps the file automatically
    with open(target_path, "w", encoding="utf-8") as f:
        f.write(tzx_data)
    print(f"[tzpro-agent] successfully injected {layer_name} layer to TimeZero.")

## 🔄 The Injection Pipeline: How to Automate It
To make a mark appear instantly from your tzpro-agent code, follow this specific background workflow:
## 1. Target the Hot Folder
TimeZero Professional constantly scans a hidden system directory for new user objects:

## 2. The Atomic Code Loop
If tzpro-agent writes directly to the Import\ directory while a file is compiling, TimeZero might try to parse a partial file and crash. To prevent this, your Python daemon must follow the Atomic Write Pattern:

   1. Compile and save the XML payload into a temporary file inside your local workspace.
   2. Use an atomic operating system operation (os.replace) to instantly drop the completed file into the TimeZero Import\ hot directory.

# tzpro-agent/src/tzx_injector.pyimport osimport tempfile
TZ_IMPORT_PATH = r"C:\ProgramData\TimeZero\Import"
def inject_tzx_safely(filename: str, xml_content: str):
    """Executes an atomic transfer to ensure TZPro never parses incomplete XML files."""
    # 1. Ensure target directory exists
    os.makedirs(TZ_IMPORT_PATH, exist_ok=True)
    
    # 2. Write to a safe temporary file on the same drive
    with tempfile.NamedTemporaryFile('w', delete=False, suffix='.tzx', dir=os.path.dirname(TZ_IMPORT_PATH)) as temp_f:
        temp_f.write(xml_content)
        temp_file_path = temp_f.name

    # 3. Atomically shift the file into the hot folder
    final_destination = os.path.join(TZ_IMPORT_PATH, filename)
    os.replace(temp_file_path, final_destination)
    print(f"[tzpro-agent] Dropped {filename} into TimeZero Import pipeline.")
