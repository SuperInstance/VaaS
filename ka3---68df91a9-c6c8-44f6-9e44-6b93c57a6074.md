## The Interview Engine: How It Works

The interview engine is the heart of the Resonance Skill. It replaces the traditional "answer generation" with a "hypothesis testing" loop:

```python
# interview_engine.py
from typing import List, Tuple, Optional

class InterviewEngine:
    """
    Before committing to a definitive response, the interview engine
    proposes lightweight hypotheses to test the operator's intent.
    """

    def __init__(self, memory_substrate: MemorySubstrate):
        self.memory = memory_substrate
        self.confidence_threshold = 0.75

    def generate_response(self, operator_input: str, context: dict) -> str:
        """
        The core interview loop.
        """
        # Step 1: Parse the operator's input
        intent_hypotheses = self._generate_hypotheses(operator_input, context)

        # Step 2: Check if we have high confidence from memory
        best_hypothesis = self._check_memory(intent_hypotheses)

        if best_hypothesis and best_hypothesis.confidence >= self.confidence_threshold:
            # High confidence — commit to a definitive response
            return self._commit_response(best_hypothesis, operator_input)

        # Step 3: Low confidence — pitch softballs
        return self._pitch_softballs(intent_hypotheses, operator_input)

    def _generate_hypotheses(self, operator_input: str, context: dict) -> List[Hypothesis]:
        """
        Generate 2-3 hypotheses about the operator's latent intent.
        """
        hypotheses = []

        # Hypothesis 1: Informational intent
        hypotheses.append(Hypothesis(
            intent="informational",
            description="The operator wants to know something",
            confidence=0.5,
            test_phrase="Are you looking for data on this, or..."
        ))

        # Hypothesis 2: Actional intent
        hypotheses.append(Hypothesis(
            intent="actional",
            description="The operator wants to do something",
            confidence=0.5,
            test_phrase="Shall I take action on this, or..."
        ))

        # Hypothesis 3: Observational intent
        hypotheses.append(Hypothesis(
            intent="observational",
            description="The operator is logging an observation",
            confidence=0.5,
            test_phrase="Shall I log this observation, or..."
        ))

        # Adjust confidence based on memory substrate
        for h in hypotheses:
            h.confidence += self.memory.intention_field.get(h.intent, 0.0)

        return hypotheses

    def _pitch_softballs(self, hypotheses: List[Hypothesis], operator_input: str) -> str:
        """
        Generate a response that pitches 2-3 lightweight options.
        """
        # Sort by confidence
        sorted_hypotheses = sorted(hypotheses, key=lambda h: h.confidence, reverse=True)

        # Build the softball pitch
        pitch = f"I hear you: '{operator_input}'\n\n"

        if len(sorted_hypotheses) >= 2:
            pitch += f"Are you thinking:\n"
            pitch += f"• {sorted_hypotheses[0].test_phrase}\n"
            pitch += f"• Or perhaps {sorted_hypotheses[1].test_phrase}\n"

        if len(sorted_hypotheses) >= 3:
            pitch += f"• Or maybe {sorted_hypotheses[2].test_phrase}\n"

        pitch += "\nWhat feels right?"

        return pitch

    def _commit_response(self, hypothesis: Hypothesis, operator_input: str) -> str:
        """
        High confidence — commit to a definitive response.
        """
        return f"Got it. Based on your pattern, I'll {hypothesis.description}."
```

```python
# resonance_detector.py
import time
from typing import Optional, Dict
from dataclasses import dataclass

@dataclass
class InteractionMetrics:
    """Metrics for a single interaction."""
    input_timestamp: float
    response_timestamp: float
    operator_latency: float  # Time between AI response and operator's next input
    response_length: int  # Length of operator's response
    engagement_depth: int  # Number of follow-up questions or clarifications
    emotional_valence: float  # -1.0 to 1.0 (detected via sentiment analysis)
    topic_persistence: bool  # Did the operator stay on topic or pivot?

class ResonanceDetector:
    """
    Monitors implicit feedback signals to detect resonance or dissonance.
    """

    def __init__(self, memory_substrate: MemorySubstrate):
        self.memory = memory_substrate
        self.last_interaction: Optional[InteractionMetrics] = None

    def analyze(self, operator_input: str, ai_response: str, context: dict) -> ResonanceMemory:
        """
        Analyze the operator's implicit feedback to the AI's response.
        """
        current_time = time.time()

        # Calculate latency
        latency = current_time - self.last_interaction.response_timestamp if self.last_interaction else 0.0

        # Calculate engagement depth
        engagement_depth = self._calculate_engagement_depth(operator_input)

        # Detect emotional valence
        valence = self._detect_valence(operator_input)

        # Check topic persistence
        topic_persistence = self._check_topic_persistence(operator_input, context)

        # Build resonance memory
        resonance = ResonanceMemory(
            timestamp=datetime.now(),
            operator_id=self.memory.operator_id,
            valence_shift=valence,
            pacing_change=latency - (self.last_interaction.operator_latency if self.last_interaction else 0.0),
            engagement_spike=engagement_depth > 2,
            topic_persistence=[context.get("current_topic", "unknown")],
            silence_pattern=latency if latency > 10.0 else None
        )

        # Update memory substrate
        self.memory.resonance.append(resonance)

        return resonance

    def _calculate_engagement_depth(self, operator_input: str) -> int:
        """Calculate how deeply the operator engaged with the response."""
        # Simple heuristic: number of questions, clarifications, or refinements
        depth = 0
        if "?" in operator_input:
            depth += 1
        if any(word in operator_input.lower() for word in ["clarify", "explain", "elaborate", "detail"]):
            depth += 1
        if len(operator_input) > 100:  # Long response = deep engagement
            depth += 1
        return depth

    def _detect_valence(self, operator_input: str) -> float:
        """Detect emotional valence of operator's response."""
        # Simple sentiment analysis
        positive_words = ["good", "great", "perfect", "exactly", "yes", "thanks", "nice"]
        negative_words = ["no", "wrong", "bad", "incorrect", "not", "stop", "don't", "ugh"]

        valence = 0.0
        for word in positive_words:
            if word in operator_input.lower():
                valence += 0.3
        for word in negative_words:
            if word in operator_input.lower():
                valence -= 0.3

        return max(-1.0, min(1.0, valence))

    def _check_topic_persistence(self, operator_input: str, context: dict) -> bool:
        """Check if the operator stayed on topic or pivoted."""
        current_topic = context.get("current_topic", "")
        # Simple check: does the input contain keywords from the current topic?
        return current_topic.lower() in operator_input.lower()
```

```python
            "verbosity": verbosity,
            "technical_depth": technical_depth,
            "formality": formality,
            "humor": "low",  # Maritime is serious — keep humor low
            "directness": directness
        }
```
